function rotate(arr) {
  let result = Array.from({ length: arr.length }, () =>
    Array(arr.length).fill(0),
  );
  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr.length; j++) {
      result[j][result.length - 1 - i] = arr[i][j];
    }
  }
  return result;
}

function rotateClock(arr, n) {
  let result;
  for (let i = 0; i <= n; i++) {
    result = rotate(arr);
  }
  return result;
}

console.log(
